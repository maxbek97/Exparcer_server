CREATE DATABASE excel_parcer;
use excel_parcer;
create Table guides (
    id_document VARCHAR(20) NOT NULL,
    name_guid VARCHAR(90) NOT NULL,
    acceptance_date DATE NOT NULL,
    check_date DATE NOT NULL,
    id_struct_unit TINYINT UNSIGNED Not NULL,
    PRIMARY KEY (id_document, id_struct_unit),
    FOREIGN KEY (id_struct_unit) REFERENCES struct_units(id_unit)
)


create table struct_units (
    id_unit TINYINT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
    name_struct VARCHAR(200) NOT NULL UNIQUE
)
